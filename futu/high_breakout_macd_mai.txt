{順勢交易：15日新高突破 + MACD(5,35,5)>0 + 20日新低出場}
{由 Pine Script strategy 轉成富途 MAI 主圖指標。}
{2026-07 回測 tune：15/20 + 淨 MACD 係穩定 plateau；200MA 濾網預設關閉。}
{注意：INITIAL_CAPITAL/DEFAULT_QTY/PYRAMIDING 屬回測交易設定，MAI 指標公式不處理。}

BREAKOUTLEN:=15;
EXITLEN:=20;
FASTLEN:=5;
SLOWLEN:=35;
SIGNALLEN:=5;
MALEN:=200;
USEMACD:=1;
USEMA:=0;

{取前一根 K 線已知的區間高低，避免當前 K 線自己計入突破位}
HIGHESTHIGH:=REF(HHV(HIGH,BREAKOUTLEN),1);
LOWESTLOW:=REF(LLV(LOW,EXITLEN),1);
MALINE:=MA(CLOSE,MALEN);

DIF:=EMA(CLOSE,FASTLEN)-EMA(CLOSE,SLOWLEN);
DEA:=EMA(DIF,SIGNALLEN);
MACDVAL:=(DIF-DEA)*2;

ISHIGHESTHIGH:=HIGH>HIGHESTHIGH;
ISLOWESTLOW:=LOW<LOWESTLOW;

{對應 Pine：isHighestHigh and (not useMacd or macdLine > 0) and (not useMa or close > ma)}
MACDOK:=USEMACD=0 OR DIF>0;
MAOK:=USEMA=0 OR CLOSE>MALINE;

{原始條件：持倉期間會重複觸發，需要用持倉狀態過濾}
RAWBUY:=ISHIGHESTHIGH AND MACDOK AND MAOK;
RAWSELL:=ISLOWESTLOW;

{模擬持倉狀態：最近一次 RAWBUY 比 RAWSELL 近，即視為持倉中}
{注意：最初 warm-up 未齊資料時可能未穩定，實際訊號以足夠 K 線後為準}
LONGSTATE:=BARSLAST(RAWBUY)<BARSLAST(RAWSELL);

{只喺狀態翻轉嗰根 K 線出訊號，避免持倉期間重複 BUY}
BUY:=LONGSTATE AND NOT(REF(LONGSTATE,1));
SELL:=NOT(LONGSTATE) AND REF(LONGSTATE,1);

新高線:HIGHESTHIGH,COLORGREEN;
新低線:LOWESTLOW,COLORRED;
MA趨勢線:MALINE,COLORBLUE;

DRAWTEXT(ISHIGHESTHIGH,HIGH,'H15'),COLORGREEN;
DRAWTEXT(ISLOWESTLOW,LOW,'L20'),COLORRED;
DRAWTEXT(BUY,LOW,'BUY'),COLORWHITE;
DRAWTEXT(SELL,HIGH,'SELL'),COLORWHITE;
